    static void imprimirResultados(Resultado[] resultados) {
        Resultado mejor = resultados[0];
        for (Resultado r : resultados) {
            System.out.printf("%-18s | Comparaciones: %-9d | Tiempo: %-10d ns | Memoria: %d bytes%n",
                    r.algoritmo, r.comparaciones, r.tiempoNs, r.memoriaBytes);
            if (r.tiempoNs < mejor.tiempoNs) mejor = r;
        }
        System.out.println("---------------------------------------------");
        System.out.println("Algoritmo mas eficiente (por tiempo): " + mejor.algoritmo);
    }

    static Resultado medirTexto(String nombre, String[] base, OrdenadorTexto metodo) {
        String[] copia = Arrays.copyOf(base, base.length);
        long memAntes = memoriaUsada();
        long inicio = System.nanoTime();
        long comparaciones = metodo.ordenar(copia);
        long fin = System.nanoTime();
        long memDespues = memoriaUsada();
        long deltaMem = Math.max(0, memDespues - memAntes);
        return new Resultado(nombre, comparaciones, fin - inicio, deltaMem);
    }

    static Resultado medirNumero(String nombre, int[] base, OrdenadorNumero metodo) {
        int[] copia = Arrays.copyOf(base, base.length);
        long memAntes = memoriaUsada();
        long inicio = System.nanoTime();
        long comparaciones = metodo.ordenar(copia);
        long fin = System.nanoTime();
        long memDespues = memoriaUsada();
        long deltaMem = Math.max(0, memDespues - memAntes);
        return new Resultado(nombre, comparaciones, fin - inicio, deltaMem);
    }

    static long burbujaSimpleTexto(String[] arr) {
        long comp = 0;
        for (int i = 0; i < arr.length - 1; i++)
            for (int j = 0; j < arr.length - 1 - i; j++) {
                comp++;
                if (compararTexto(arr[j], arr[j + 1]) > 0) {
                    String t = arr[j]; arr[j] = arr[j + 1]; arr[j + 1] = t;
                }
            }
        return comp;
    }

    static long burbujaMejoradoTexto(String[] arr) {
        long comp = 0;
        boolean huboCambio;
        for (int i = 0; i < arr.length - 1; i++) {
            huboCambio = false;
            for (int j = 0; j < arr.length - 1 - i; j++) {
                comp++;
                if (compararTexto(arr[j], arr[j + 1]) > 0) {
                    String t = arr[j]; arr[j] = arr[j + 1]; arr[j + 1] = t;
                    huboCambio = true;
                }
            }
            if (!huboCambio) break;
        }
        return comp;
    }

    static long quickSortTexto(String[] arr) {
        long[] comp = {0};
        quickSortRecTexto(arr, 0, arr.length - 1, comp);
        return comp[0];
    }

    static void quickSortRecTexto(String[] arr, int low, int high, long[] comp) {
        if (low < high) {
            int p = particionTexto(arr, low, high, comp);
            quickSortRecTexto(arr, low, p - 1, comp);
            quickSortRecTexto(arr, p + 1, high, comp);
        }
    }

    static int particionTexto(String[] arr, int low, int high, long[] comp) {
        String pivote = arr[high];
        int i = low - 1;
        for (int j = low; j < high; j++) {
            comp[0]++;
            if (compararTexto(arr[j], pivote) <= 0) {
                i++;
                String t = arr[i]; arr[i] = arr[j]; arr[j] = t;
            }
        }
        String t = arr[i + 1]; arr[i + 1] = arr[high]; arr[high] = t;
        return i + 1;
    }

    static long shellSortTexto(String[] arr) {
        long comp = 0;
        for (int gap = arr.length / 2; gap > 0; gap /= 2)
            for (int i = gap; i < arr.length; i++) {
                String temp = arr[i];
                int j = i;
                while (j >= gap) {
                    comp++;
                    if (compararTexto(arr[j - gap], temp) > 0) {
                        arr[j] = arr[j - gap];
                        j -= gap;
                    } else break;
                }
                arr[j] = temp;
            }
        return comp;
    }

    static long seleccionDirectaTexto(String[] arr) {
        long comp = 0;
        for (int i = 0; i < arr.length - 1; i++) {
            int min = i;
            for (int j = i + 1; j < arr.length; j++) {
                comp++;
                if (compararTexto(arr[j], arr[min]) < 0) min = j;
            }
            String t = arr[i]; arr[i] = arr[min]; arr[min] = t;
        }
        return comp;
    }

    static long insercionTexto(String[] arr) {
        long comp = 0;
        for (int i = 1; i < arr.length; i++) {
            String clave = arr[i];
            int j = i - 1;
            while (j >= 0) {
                comp++;
                if (compararTexto(arr[j], clave) > 0) {
                    arr[j + 1] = arr[j];
                    j--;
                } else break;
            }
            arr[j + 1] = clave;
        }
        return comp;
    }

    static long quickSortNum(int[] arr) {
        long[] comp = {0};
        quickSortRecNum(arr, 0, arr.length - 1, comp);
        return comp[0];
    }

    static void quickSortRecNum(int[] arr, int low, int high, long[] comp) {
        if (low < high) {
            int p = particionNum(arr, low, high, comp);
            quickSortRecNum(arr, low, p - 1, comp);
            quickSortRecNum(arr, p + 1, high, comp);
        }
    }

    static int particionNum(int[] arr, int low, int high, long[] comp) {
        int pivote = arr[high];
        int i = low - 1;
        for (int j = low; j < high; j++) {
            comp[0]++;
            if (arr[j] <= pivote) {
                i++;
                int t = arr[i]; arr[i] = arr[j]; arr[j] = t;
            }
        }
        int t = arr[i + 1]; arr[i + 1] = arr[high]; arr[high] = t;
        return i + 1;
    }

    static long shellSortNum(int[] arr) {
        long comp = 0;
        for (int gap = arr.length / 2; gap > 0; gap /= 2)
            for (int i = gap; i < arr.length; i++) {
                int temp = arr[i];
                int j = i;
                while (j >= gap) {
                    comp++;
                    if (arr[j - gap] > temp) {
                        arr[j] = arr[j - gap];
                        j -= gap;
                    } else break;
                }
                arr[j] = temp;
            }
        return comp;
    }

    static long seleccionDirectaNum(int[] arr) {
        long comp = 0;
        for (int i = 0; i < arr.length - 1; i++) {
            int min = i;
            for (int j = i + 1; j < arr.length; j++) {
                comp++;
                if (arr[j] < arr[min]) min = j;
            }
            int t = arr[i]; arr[i] = arr[min]; arr[min] = t;
        }
        return comp;
    }

    static long insercionNum(int[] arr) {
        long comp = 0;
        for (int i = 1; i < arr.length; i++) {
            int clave = arr[i];
            int j = i - 1;
            while (j >= 0) {
                comp++;
                if (arr[j] > clave) {
                    arr[j + 1] = arr[j];
                    j--;
                } else break;
            }
            arr[j + 1] = clave;
        }
        return comp;
    }
