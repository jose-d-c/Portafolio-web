// =============================================================================
// Escenarios cubiertos
//  - Cantidades: 15, 100 y 500 elementos.
//  - Distribuciones: datos limitados, semi-aleatorios y totalmente invertidos.
//  - Metricas: iteraciones, comparaciones estimadas, movimientos, tiempo y memoria.
//  - Se demuestra estabilidad reconstruyendo pares (valor, etiqueta) y verificando
//    que las etiquetas originales no se mezclan.
// =============================================================================

import java.time.Duration;
import java.time.Instant;
import java.util.Arrays;
import java.util.Locale;
import java.util.Random;

public class Programa29CountingSort {

    private static final Random RANDOM = new Random();

    public static void main(String[] args) {
        Locale.setDefault(Locale.US);
        ejecutarEscenario("Datos bajos", generarDatos(15, 15));
        ejecutarEscenario("Datos medios", generarDatos(100, 120));
        ejecutarEscenario("Datos altos", generarDatos(500, 256));
        ejecutarEscenario("Datos invertidos", generarInvertidos(120));
    }

    private static void ejecutarEscenario(String etiqueta, int[] datos) {
        Instrumentacion instrumentacion = new Instrumentacion();
        int rangoMaximo = Arrays.stream(datos).max().orElse(0) + 1;

        Instant inicio = Instant.now();
        int[][] resultado = countingSortConEtiquetas(datos, rangoMaximo, instrumentacion);
        Instant fin = Instant.now();

        instrumentacion.tiempoNs = Duration.between(inicio, fin).toNanos();

        System.out.println("======================================================");
        System.out.println("ESCENARIO: " + etiqueta);
        System.out.println("ENTRADA: " + Arrays.toString(Arrays.copyOf(datos, Math.min(datos.length, 30))));
        System.out.printf("Iteraciones datos: %d\n", instrumentacion.iteracionesDatos);
        System.out.printf("Iteraciones contador: %d\n", instrumentacion.iteracionesContador);
        System.out.printf("Comparaciones (estimadas por reconstruccion): %d\n", instrumentacion.comparaciones);
        System.out.printf("Movimientos totales: %d\n", instrumentacion.movimientos);
        System.out.printf("Tiempo (ms): %.4f\n", instrumentacion.tiempoNs / 1_000_000.0);
        System.out.printf("Memoria estimada (bytes): %d\n", instrumentacion.memoriaBytes);
        System.out.println("Salida ordenada (primeros valores): " + Arrays.toString(Arrays.stream(resultado)
                .limit(30)
                .mapToInt(par -> par[0])
                .toArray()));
        System.out.println("Estabilidad validada?: " + validarEstabilidad(resultado));
    }

    private static int[][] countingSortConEtiquetas(int[] datos, int rango, Instrumentacion instrumentacion) {
        int[][] pares = new int[datos.length][2];
        for (int i = 0; i < datos.length; i++) {
            pares[i][0] = datos[i];    // valor
            pares[i][1] = i;           // etiqueta original
        }

        int[] contador = new int[rango];
        instrumentacion.memoriaBytes = (long) (rango * Integer.BYTES) + (long) (datos.length * Integer.BYTES * 2);

        for (int[] par : pares) {
            instrumentacion.iteracionesDatos++;
            contador[par[0]]++;
        }

        for (int i = 1; i < contador.length; i++) {
            instrumentacion.iteracionesContador++;
            contador[i] += contador[i - 1];
        }

        int[][] salida = new int[datos.length][2];
        for (int i = pares.length - 1; i >= 0; i--) {
            int valor = pares[i][0];
            int posicion = --contador[valor];
            salida[posicion][0] = valor;
            salida[posicion][1] = pares[i][1];
            instrumentacion.movimientos++;
        }

        // Contar comparaciones aproximadas: cada lectura del contador + verificacion de rango
        instrumentacion.comparaciones = instrumentacion.iteracionesDatos + instrumentacion.iteracionesContador;
        return salida;
    }

    private static boolean validarEstabilidad(int[][] paresOrdenados) {
        for (int i = 1; i < paresOrdenados.length; i++) {
            if (paresOrdenados[i - 1][0] == paresOrdenados[i][0]) {
                if (paresOrdenados[i - 1][1] > paresOrdenados[i][1]) {
                    return false;
                }
            }
        }
        return true;
    }

    private static int[] generarDatos(int cantidad, int maxValor) {
        int[] datos = new int[cantidad];
        for (int i = 0; i < cantidad; i++) {
            datos[i] = RANDOM.nextInt(maxValor);
        }
        return datos;
    }

    private static int[] generarInvertidos(int cantidad) {
        int[] datos = new int[cantidad];
        for (int i = 0; i < cantidad; i++) {
            datos[i] = cantidad - i;
        }
        return datos;
    }

    private static final class Instrumentacion {
        long iteracionesDatos;
        long iteracionesContador;
        long comparaciones;
        long movimientos;
        long memoriaBytes;
        long tiempoNs;
    }
}
