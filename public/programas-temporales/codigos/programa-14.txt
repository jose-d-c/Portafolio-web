// ============================================================================
// PROGRAMA 13 - COMPARACION DE MEMORIA DE VARIOS ALGORITMOS DE ORDENAMIENTO
// ============================================================================
// EXPLICACION MUY SIMPLE:
// Este programa no se concentra en un solo algoritmo.
// Aqui ponemos a competir varios algoritmos de ordenamiento usando exactamente
// las mismas situaciones de prueba para poder compararlos de manera justa.
//
// QUE SE PRUEBA?
// - 3 tamanos: 10, 50 y 200 elementos.
// - 3 estados iniciales: totalmente desorganizados, parcialmente organizados
//   y ordenados invertidamente.
// - 6 algoritmos: Burbuja, Burbuja Mejorado, QuickSort, ShellSort,
//   Seleccion e Insercion.
//
// QUE MIDE ESTE PROGRAMA?
// - MEMORIA USADA en bytes, sin dejar de mostrar otras metricas utiles.
//
// IDEA PEDAGOGICA:
// Si todos reciben los mismos datos, se puede observar con mas justicia cual
// tarda mas y cual tarda menos segun el escenario.
// ============================================================================

import java.util.Arrays;

public class Programa14ComparacionMemoria {

    static class ResultadoMemoria {
        long memoriaBytes;
        long comparaciones;
        long intercambios;
        long iteraciones;
    }

    public static void main(String[] args) {
        ejecutarEscenario("POCOS DATOS (10)", "TOTALMENTE DESORGANIZADOS", generarEnterosDesordenados(10));
        ejecutarEscenario("POCOS DATOS (10)", "PARCIALMENTE ORGANIZADOS", generarEnterosParcialmenteOrdenados(10));
        ejecutarEscenario("POCOS DATOS (10)", "ORDENADOS INVERTIDAMENTE", generarEnterosInvertidos(10));

        ejecutarEscenario("DATOS MODERADOS (50)", "TOTALMENTE DESORGANIZADOS", generarEnterosDesordenados(50));
        ejecutarEscenario("DATOS MODERADOS (50)", "PARCIALMENTE ORGANIZADOS", generarEnterosParcialmenteOrdenados(50));
        ejecutarEscenario("DATOS MODERADOS (50)", "ORDENADOS INVERTIDAMENTE", generarEnterosInvertidos(50));

        ejecutarEscenario("GRAN CANTIDAD DE DATOS (200)", "TOTALMENTE DESORGANIZADOS", generarEnterosDesordenados(200));
        ejecutarEscenario("GRAN CANTIDAD DE DATOS (200)", "PARCIALMENTE ORGANIZADOS", generarEnterosParcialmenteOrdenados(200));
        ejecutarEscenario("GRAN CANTIDAD DE DATOS (200)", "ORDENADOS INVERTIDAMENTE", generarEnterosInvertidos(200));
    }

    private static void ejecutarEscenario(String tamano, String estado, int[] base) {
        System.out.println("============================================================");
        System.out.println("TAMANO: " + tamano);
        System.out.println("ESTADO: " + estado);
        System.out.println("MUESTRA BASE: " + resumir(base, 20));
        medirYMostrar("Burbuja", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::burbuja);
        medirYMostrar("Burbuja Mejorado", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::burbujaMejorado);
        medirYMostrar("QuickSort", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::quickSort);
        medirYMostrar("ShellSort", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::shellSort);
        medirYMostrar("Seleccion", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::seleccion);
        medirYMostrar("Insercion", Arrays.copyOf(base, base.length), Programa14ComparacionMemoria::insercion);
        System.out.println();
    }

    interface Algoritmo { ResultadoMemoria aplicar(int[] datos); }

    private static void medirYMostrar(String nombre, int[] datos, Algoritmo algoritmo) {
        ResultadoMemoria r = algoritmo.aplicar(datos);
        System.out.println("--- " + nombre + " ---");
        System.out.println("MEMORIA (bytes): " + r.memoriaBytes);
        System.out.println("COMPARACIONES: " + r.comparaciones);
        System.out.println("INTERCAMBIOS/MOVIMIENTOS: " + r.intercambios);
        System.out.println("ITERACIONES: " + r.iteraciones);
    }

    private static ResultadoMemoria burbuja(int[] a) {
        ResultadoMemoria r = new ResultadoMemoria();
        Runtime runtime = Runtime.getRuntime();
        runtime.gc();
        long iniMem = runtime.totalMemory() - runtime.freeMemory();
        for (int i=0;i<a.length-1;i++) { r.iteraciones++; for (int j=0;j<a.length-1-i;j++) { r.comparaciones++; if (a[j]>a[j+1]) { int t=a[j]; a[j]=a[j+1]; a[j+1]=t; r.intercambios++; } } }
        runtime.gc();
        long finMem = runtime.totalMemory() - runtime.freeMemory();
        r.memoriaBytes = Math.max(finMem - iniMem, 0);
        return r;
    }
    private static ResultadoMemoria burbujaMejorado(int[] a) {
        ResultadoMemoria r = new ResultadoMemoria();
        Runtime runtime = Runtime.getRuntime();
        runtime.gc();
        long iniMem = runtime.totalMemory() - runtime.freeMemory();
        for (int i=0;i<a.length-1;i++) { boolean cambio=false; r.iteraciones++; for (int j=0;j<a.length-1-i;j++) { r.comparaciones++; if (a[j]>a[j+1]) { int t=a[j]; a[j]=a[j+1]; a[j+1]=t; r.intercambios++; cambio=true; } } if(!cambio) break; }
        runtime.gc();
        long finMem = runtime.totalMemory() - runtime.freeMemory();
        r.memoriaBytes = Math.max(finMem - iniMem, 0);
        return r;
    }
    private static ResultadoMemoria quickSort(int[] a) { ResultadoMemoria r = new ResultadoMemoria(); Runtime runtime=Runtime.getRuntime(); runtime.gc(); long iniMem=runtime.totalMemory()-runtime.freeMemory(); long ini=System.nanoTime(); quickSortRec(a,0,a.length-1,r); runtime.gc(); long finMem=runtime.totalMemory()-runtime.freeMemory(); r.memoriaBytes=Math.max(finMem-iniMem,0); return r; }
    private static void quickSortRec(int[] a,int izq,int der,ResultadoMemoria r){ r.iteraciones++; if(izq>=der)return; int piv=a[der],i=izq-1; for(int j=izq;j<der;j++){ r.comparaciones++; if(a[j]<=piv){ i++; int t=a[i]; a[i]=a[j]; a[j]=t; r.intercambios++; } } int t=a[i+1]; a[i+1]=a[der]; a[der]=t; r.intercambios++; int p=i+1; quickSortRec(a,izq,p-1,r); quickSortRec(a,p+1,der,r);} 
    private static ResultadoMemoria shellSort(int[] a) { ResultadoMemoria r = new ResultadoMemoria(); Runtime runtime=Runtime.getRuntime(); runtime.gc(); long iniMem=runtime.totalMemory()-runtime.freeMemory(); long ini=System.nanoTime(); for(int gap=a.length/2; gap>0; gap/=2){ for(int i=gap;i<a.length;i++){ r.iteraciones++; int temp=a[i], j=i; while(j>=gap){ r.comparaciones++; if(a[j-gap]>temp){ a[j]=a[j-gap]; r.intercambios++; j-=gap;} else break;} a[j]=temp; r.intercambios++; } } runtime.gc(); long finMem=runtime.totalMemory()-runtime.freeMemory(); r.memoriaBytes=Math.max(finMem-iniMem,0); return r; }
    private static ResultadoMemoria seleccion(int[] a) { ResultadoMemoria r = new ResultadoMemoria(); Runtime runtime=Runtime.getRuntime(); runtime.gc(); long iniMem=runtime.totalMemory()-runtime.freeMemory(); long ini=System.nanoTime(); for(int i=0;i<a.length-1;i++){ r.iteraciones++; int min=i; for(int j=i+1;j<a.length;j++){ r.comparaciones++; if(a[j]<a[min]) min=j; } if(min!=i){ int t=a[i]; a[i]=a[min]; a[min]=t; r.intercambios++; } } runtime.gc(); long finMem=runtime.totalMemory()-runtime.freeMemory(); r.memoriaBytes=Math.max(finMem-iniMem,0); return r; }
    private static ResultadoMemoria insercion(int[] a) { ResultadoMemoria r = new ResultadoMemoria(); Runtime runtime=Runtime.getRuntime(); runtime.gc(); long iniMem=runtime.totalMemory()-runtime.freeMemory(); long ini=System.nanoTime(); for(int i=1;i<a.length;i++){ r.iteraciones++; int clave=a[i],j=i-1; while(j>=0){ r.comparaciones++; if(a[j]>clave){ a[j+1]=a[j]; r.intercambios++; j--; } else break; } a[j+1]=clave; r.intercambios++; } runtime.gc(); long finMem=runtime.totalMemory()-runtime.freeMemory(); r.memoriaBytes=Math.max(finMem-iniMem,0); return r; }

    private static int[] generarEnterosInvertidos(int n){ int[] d=new int[n]; for(int i=0;i<n;i++) d[i]=n-i; return d; }
    private static int[] generarEnterosParcialmenteOrdenados(int n){ int[] d=generarEnterosAscendentes(n); if(n>=4){ intercambiar(d,n/4,n/4+1); intercambiar(d,n/2,Math.min(n-1,n/2+2)); intercambiar(d,Math.max(0,n-3),n-1);} return d; }
    private static int[] generarEnterosDesordenados(int n){ int[] d=new int[n]; for(int i=0;i<n;i++) d[i]=((i*37)+11)%(n*3+17)+1; return d; }
    private static int[] generarEnterosAscendentes(int n){ int[] d=new int[n]; for(int i=0;i<n;i++) d[i]=i+1; return d; }
    private static void intercambiar(int[] a,int i,int j){ int t=a[i]; a[i]=a[j]; a[j]=t; }
    private static String resumir(int[] a,int limite){ return a.length<=limite?Arrays.toString(a):Arrays.toString(Arrays.copyOf(a, limite))+" ... (total "+a.length+" elementos)"; }
}
