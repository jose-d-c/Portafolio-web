    // Estructura para guardar las metricas de cada algoritmo.
    // comparaciones: cuantas veces se comparan dos elementos.
    // movimientos: asignaciones/intercambios/desplazamientos realizados.
    // tiempoNs: tiempo total de ejecucion en nanosegundos.
    static class Metricas {
        long iteraciones;
        long comparaciones;
        long intercambios; // swaps reales entre dos posiciones
        long movimientos;  // asignaciones/desplazamientos
    }

    static class Resultado {
        String algoritmo;
        long iteraciones;
        long comparaciones;
        long intercambios;
        long movimientos;
        long tiempoNs;
        long memoriaBytes;
        boolean ok;

        Resultado(String algoritmo, Metricas m, long tiempoNs, long memoriaBytes, boolean ok) {
            this.algoritmo = algoritmo;
            this.iteraciones = m.iteraciones;
            this.comparaciones = m.comparaciones;
            this.intercambios = m.intercambios;
            this.movimientos = m.movimientos;
            this.tiempoNs = tiempoNs;
            this.memoriaBytes = memoriaBytes;
            this.ok = ok;
        }
    }

    interface Ordenador<T extends Comparable<T>> {
        Metricas ejecutar(T[] arr);
    }



    static class ConteoNombre implements Comparable<ConteoNombre> {
        String nombre;
        int repeticiones;
        static boolean ordenarPorCantidad = false;

        ConteoNombre(String nombre, int repeticiones) {
            this.nombre = nombre;
            this.repeticiones = repeticiones;
        }

        @Override
        public int compareTo(ConteoNombre o) {
            if (ordenarPorCantidad) {
                int c = Integer.compare(o.repeticiones, this.repeticiones); // descendente
                if (c != 0) return c;
                return this.nombre.compareTo(o.nombre);
            }
            return this.nombre.compareTo(o.nombre);
        }

        @Override
        public String toString() {
            return nombre + " -> " + repeticiones;
        }
    }

    static class RegistroDolar implements Comparable<RegistroDolar> {
        String periodo;
        double valor;

        RegistroDolar(String periodo, double valor) {
            this.periodo = periodo;
            this.valor = valor;
        }

        @Override
        public int compareTo(RegistroDolar o) {
            return Double.compare(this.valor, o.valor);
        }

        @Override
        public String toString() {
            return periodo + "=" + valor;
        }
    }

    static <T> T[] copiarInvertido(T[] base) {
        T[] copia = Arrays.copyOf(base, base.length);
        for (int i = 0, j = copia.length - 1; i < j; i++, j--) {
            T tmp = copia[i];
            copia[i] = copia[j];
            copia[j] = tmp;
        }
        return copia;
    }

    // Concatena dos arreglos del mismo tipo en uno solo.
    static <T> T[] concatenar(T[] a, T[] b) {
        T[] out = Arrays.copyOf(a, a.length + b.length);
        System.arraycopy(b, 0, out, a.length, b.length);
        return out;
    }

    static <T> void barajar(T[] arr, long semilla) {
        Random r = new Random(semilla);
        for (int i = arr.length - 1; i > 0; i--) {
            int j = r.nextInt(i + 1);
            T t = arr[i]; arr[i] = arr[j]; arr[j] = t;
        }
    }

    static Map<String, Integer> contarRepeticiones(String[] nombres) {
        Map<String, Integer> frecuencias = new HashMap<>();
        for (String n : nombres) {

    static <T extends Comparable<T>> T[] ordenarYRetornar(T[] base, String algoritmo) {
        T[] copia = Arrays.copyOf(base, base.length);
        switch (algoritmo) {
            case "Burbuja Simple" -> burbujaSimple(copia);
            case "Burbuja Mejorado" -> burbujaMejorado(copia);
            case "Quick Sort" -> quickSortWrapper(copia);
            case "Shell Sort" -> shellSort(copia);
            case "Seleccion Directa" -> seleccionDirecta(copia);
            default -> insercion(copia);
        }
        return copia;
    }

    // Ejecuta los seis algoritmos sobre una copia del arreglo y muestra tabla.
    static <T extends Comparable<T>> void ejecutarEscenario(String nombreEscenario, T[] base) {
        Map<String, Resultado> resultados = new LinkedHashMap<>();

        resultados.put("Burbuja Simple", medir("Burbuja Simple", base, ProgramaTemporal18::burbujaSimple));
        resultados.put("Burbuja Mejorado", medir("Burbuja Mejorado", base, ProgramaTemporal18::burbujaMejorado));
        resultados.put("Quick Sort", medir("Quick Sort", base, ProgramaTemporal18::quickSortWrapper));
        resultados.put("Shell Sort", medir("Shell Sort", base, ProgramaTemporal18::shellSort));
        resultados.put("Seleccion Directa", medir("Seleccion Directa", base, ProgramaTemporal18::seleccionDirecta));
        resultados.put("Insercion", medir("Insercion", base, ProgramaTemporal18::insercion));

        System.out.println("Escenario: " + nombreEscenario + " (n=" + base.length + ")");
        imprimirResultados(resultados);
    }

    static <T extends Comparable<T>> Resultado medir(String nombre, T[] original, Ordenador<T> metodo) {
        T[] copia = Arrays.copyOf(original, original.length);
        Runtime rt = Runtime.getRuntime();
        rt.gc();
        long memAntes = rt.totalMemory() - rt.freeMemory();
        long inicio = System.nanoTime();
        Metricas m = metodo.ejecutar(copia);
        long fin = System.nanoTime();
        long memDespues = rt.totalMemory() - rt.freeMemory();
        long memoria = Math.max(0, memDespues - memAntes);
        boolean ok = estaOrdenado(copia);
        if (!ok) {
            System.out.println("ERROR: no ordeno correctamente -> " + nombre);
        }
        return new Resultado(nombre, m, fin - inicio, memoria, ok);
    }

    static <T extends Comparable<T>> boolean estaOrdenado(T[] arr) {
        for (int i = 1; i < arr.length; i++) {
            if (arr[i - 1].compareTo(arr[i]) > 0) return false;
        }
        return true;
    }

    // Burbuja simple: compara pares adyacentes e intercambia si estan invertidos.
    static <T extends Comparable<T>> Metricas burbujaSimple(T[] arr) {
        Metricas m = new Metricas();
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            m.iteraciones++;
            for (int j = 0; j < n - 1 - i; j++) {
                m.iteraciones++;
                m.comparaciones++;
                if (arr[j].compareTo(arr[j + 1]) > 0) {
                    T tmp = arr[j]; m.movimientos++;
                    arr[j] = arr[j + 1]; m.movimientos++;
                    arr[j + 1] = tmp; m.movimientos++;
                    m.intercambios++;
                }
            }
        }
        return m;
    }

    // Burbuja mejorado: si una pasada no produce cambios, se detiene temprano.
    static <T extends Comparable<T>> Metricas burbujaMejorado(T[] arr) {
        Metricas m = new Metricas();
        int n = arr.length;
        boolean cambio;
        for (int i = 0; i < n - 1; i++) {
            m.iteraciones++;
            cambio = false;
            for (int j = 0; j < n - 1 - i; j++) {
                m.iteraciones++;
                m.comparaciones++;
                if (arr[j].compareTo(arr[j + 1]) > 0) {
                    T tmp = arr[j]; m.movimientos++;
                    arr[j] = arr[j + 1]; m.movimientos++;
                    arr[j + 1] = tmp; m.movimientos++;
                    m.intercambios++;
                    cambio = true;
                }
            }
            if (!cambio) break;
        }
        return m;
    }

    // Wrapper para Quick Sort recursivo.
    static <T extends Comparable<T>> Metricas quickSortWrapper(T[] arr) {
        Metricas m = new Metricas();
        quickSort(arr, 0, arr.length - 1, m);
        return m;
    }

    // Quick Sort seguro: controla limites i<=der y j>=izq antes de acceder.
    static <T extends Comparable<T>> void quickSort(T[] arr, int izq, int der, Metricas m) {
        int i = izq, j = der;
        T pivote = arr[(izq + der) / 2];

        while (i <= j) {
            m.iteraciones++;
            while (i <= der) {
                m.iteraciones++;
                m.comparaciones++;
                if (!(arr[i].compareTo(pivote) < 0)) break;
                i++;
            }
            while (j >= izq) {
                m.iteraciones++;
                m.comparaciones++;
                if (!(arr[j].compareTo(pivote) > 0)) break;
                j--;
            }
            m.comparaciones++;
            if (i <= j) {
                T temp = arr[i]; m.movimientos++;
                arr[i] = arr[j]; m.movimientos++;
                arr[j] = temp; m.movimientos++;
                m.intercambios++;
                i++;
                j--;
            }
        }

        if (izq < j) quickSort(arr, izq, j, m);
        if (i < der) quickSort(arr, i, der, m);
    }

    // Shell Sort: insercion con saltos decrecientes hasta gap=1.
    static <T extends Comparable<T>> Metricas shellSort(T[] arr) {
        Metricas m = new Metricas();
        int n = arr.length;
        for (int gap = n / 2; gap > 0; gap /= 2) {
            m.iteraciones++;
            for (int i = gap; i < n; i++) {
                m.iteraciones++;
                T temp = arr[i]; m.movimientos++;
                int j = i;
                while (j >= gap) {
                    m.iteraciones++;
                    m.comparaciones++;
                    if (!(arr[j - gap].compareTo(temp) > 0)) break;
                    arr[j] = arr[j - gap]; m.movimientos++;
                    j -= gap;
                }
                arr[j] = temp; m.movimientos++;
            }
        }
        return m;
    }

    // Seleccion directa: busca el minimo en la parte no ordenada y lo coloca adelante.
    static <T extends Comparable<T>> Metricas seleccionDirecta(T[] arr) {
        Metricas m = new Metricas();
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            m.iteraciones++;
            int min = i;
            for (int j = i + 1; j < n; j++) {
                m.iteraciones++;
                m.comparaciones++;
                if (arr[j].compareTo(arr[min]) < 0) min = j;
            }
            if (min != i) {
                T tmp = arr[i]; m.movimientos++;
                arr[i] = arr[min]; m.movimientos++;
                arr[min] = tmp; m.movimientos++;
                m.intercambios++;
            }
        }
        return m;
    }

    // Insercion: inserta cada elemento en su posicion correcta de la zona ordenada.
    static <T extends Comparable<T>> Metricas insercion(T[] arr) {
        Metricas m = new Metricas();
        int n = arr.length;
        for (int i = 1; i < n; i++) {
            m.iteraciones++;
            T clave = arr[i]; m.movimientos++;
            int j = i - 1;
            while (j >= 0) {
                m.iteraciones++;
                m.comparaciones++;
                if (!(arr[j].compareTo(clave) > 0)) break;
                arr[j + 1] = arr[j]; m.movimientos++;
                j--;
            }
            arr[j + 1] = clave; m.movimientos++;
        }
        return m;
    }
