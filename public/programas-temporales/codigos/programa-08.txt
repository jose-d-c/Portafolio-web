// 1) Este QuickSort trabaja con cadenas.
// 2) El pivote se compara con cada palabra usando compareTo.
// 3) Las palabras que van antes se agrupan a la izquierda.
// 4) Las que van despues quedan a la derecha.
// 5) El algoritmo repite el proceso en subarreglos hasta ordenar todo.
// 6) Cada comentario interno explica la logica en terminos simples.

// ================================================
// PROGRAMA 08
// Titulo: Quicksort en Java (String)
// Descripcion:
//   Ordena un arreglo de cadenas alfabeticamente con Quicksort.
//   Utiliza compareTo para la comparacion lexicografica y pivote final.
// Entradas:
//   String[] palabras = {"kiwi", "manzana", "pera", "uva", "banana", "cereza", "durazno"}
// Salidas esperadas:
//   Arreglo original: [kiwi, manzana, pera, uva, banana, cereza, durazno]
//   Arreglo ordenado: [banana, cereza, durazno, kiwi, manzana, pera, uva]
// Complejidad:
//   Promedio: O(n log n) | Peor caso: O(n^2)
// ================================================

public class Programa08QuickSortStrings {

    public static void main(String[] args) {
        String[] palabras = {"kiwi", "manzana", "pera", "uva", "banana", "cereza", "durazno"};

        System.out.print("Arreglo original: ");
        imprimir(palabras);

        quickSort(palabras, 0, palabras.length - 1);

        System.out.print("Arreglo ordenado: ");
        imprimir(palabras);
    }

    public static void quickSort(String[] arreglo, int izquierda, int derecha) {
        if (izquierda >= derecha) return;

        int indicePivote = particionar(arreglo, izquierda, derecha);
        quickSort(arreglo, izquierda, indicePivote - 1);
        quickSort(arreglo, indicePivote + 1, derecha);
    }

    public static int particionar(String[] arreglo, int izquierda, int derecha) {
        String pivote = arreglo[derecha];
        int i = izquierda - 1;

        for (int j = izquierda; j < derecha; j++) {
            if (arreglo[j].compareTo(pivote) <= 0) {
                i++;
                intercambiar(arreglo, i, j);
            }
        }

        intercambiar(arreglo, i + 1, derecha);
        return i + 1;
    }

    public static void intercambiar(String[] arreglo, int a, int b) {
        String temporal = arreglo[a];
        arreglo[a] = arreglo[b];
        arreglo[b] = temporal;
    }

    public static void imprimir(String[] arreglo) {
        System.out.print("[");
        for (int i = 0; i < arreglo.length; i++) {
            System.out.print(arreglo[i]);
            if (i < arreglo.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
}
