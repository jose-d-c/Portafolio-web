// PROGRAMA TEMPORAL 28: SIMULADOR DE BUSQUEDA BINARIA
// =============================================================================
// Objetivo:
//  - Permitir que el usuario configure tamano del arreglo, tipo de dato (int o String)
//    y valor objetivo, luego observar paso a paso como la busqueda binaria divide el
//    intervalo hasta encontrar (o descartar) el elemento.
//  - Reportar numero de comparaciones, profundidad recursiva/maxima, tiempo y memoria.
// =============================================================================

import java.text.DecimalFormat;
import java.time.Duration;
import java.time.Instant;
import java.util.Arrays;
import java.util.Locale;
import java.util.Random;
import java.util.Scanner;

public class Programa28BusquedaBinaria {

    private static final DecimalFormat FORMAT = new DecimalFormat("0.0000");
    private static final Random RANDOM = new Random();

    record ResultadoBusqueda(boolean encontrado, int posicion, int comparaciones,
                             int profundidadMaxima, long tiempoNs, long memoriaBytes) {}

    public static void main(String[] args) {
        Locale.setDefault(Locale.US);
        Scanner scanner = new Scanner(System.in);

        System.out.println("================ SIMULADOR BINARIO ================");
        System.out.print("Cantidad de elementos (entre 8 y 2000): ");
        int cantidad = Math.max(8, Math.min(2000, scanner.nextInt()));
        scanner.nextLine();

        System.out.print("Tipo de dato (int/string): ");
        String tipo = scanner.nextLine().trim().toLowerCase();

        System.out.print("Valor objetivo (numero o palabra segun el tipo): ");
        String objetivoEntrada = scanner.nextLine().trim();

        if ("string".equals(tipo)) {
            String[] datos = generarCadenasOrdenadas(cantidad);
            Arrays.sort(datos, String.CASE_INSENSITIVE_ORDER);
            ResultadoBusqueda resultado = ejecutarBusqueda(datos, objetivoEntrada.toLowerCase());
            imprimirResumen(resultado, Arrays.toString(Arrays.copyOf(datos, Math.min(datos.length, 50))));
        } else {
            int[] datos = generarEnterosOrdenados(cantidad);
            Arrays.sort(datos);
            int objetivo = parseEntero(objetivoEntrada);
            ResultadoBusqueda resultado = ejecutarBusqueda(datos, objetivo);
            imprimirResumen(resultado, Arrays.toString(Arrays.copyOf(datos, Math.min(datos.length, 50))));
        }
    }

    private static ResultadoBusqueda ejecutarBusqueda(int[] arreglo, int objetivo) {
        Runtime runtime = Runtime.getRuntime();
        runtime.gc();
        long memoriaAntes = runtime.totalMemory() - runtime.freeMemory();
        Instant inicio = Instant.now();

        Contador contador = new Contador();
        int posicion = busquedaBinaria(arreglo, objetivo, 0, arreglo.length - 1, 1, contador);

        Instant fin = Instant.now();
        runtime.gc();
        long memoriaDespues = runtime.totalMemory() - runtime.freeMemory();

        return new ResultadoBusqueda(
                posicion != -1,
                posicion,
                contador.comparaciones,
                contador.profundidadMaxima,
                Duration.between(inicio, fin).toNanos(),
                Math.max(0, memoriaDespues - memoriaAntes)
        );
    }

    private static ResultadoBusqueda ejecutarBusqueda(String[] arreglo, String objetivo) {
        Runtime runtime = Runtime.getRuntime();
        runtime.gc();
        long memoriaAntes = runtime.totalMemory() - runtime.freeMemory();
        Instant inicio = Instant.now();

        Contador contador = new Contador();
        int posicion = busquedaBinaria(arreglo, objetivo, 0, arreglo.length - 1, 1, contador);

        Instant fin = Instant.now();
        runtime.gc();
        long memoriaDespues = runtime.totalMemory() - runtime.freeMemory();

        return new ResultadoBusqueda(
                posicion != -1,
                posicion,
                contador.comparaciones,
                contador.profundidadMaxima,
                Duration.between(inicio, fin).toNanos(),
                Math.max(0, memoriaDespues - memoriaAntes)
        );
    }

    private static int busquedaBinaria(int[] arreglo, int objetivo, int inicio, int fin, int nivel, Contador contador) {
        contador.profundidadMaxima = Math.max(contador.profundidadMaxima, nivel);
        if (inicio > fin) {
            return -1;
        }
        int medio = (inicio + fin) >>> 1;
        contador.comparaciones++;
        if (arreglo[medio] == objetivo) {
            return medio;
        }
        if (arreglo[medio] > objetivo) {
            return busquedaBinaria(arreglo, objetivo, inicio, medio - 1, nivel + 1, contador);
        }
        return busquedaBinaria(arreglo, objetivo, medio + 1, fin, nivel + 1, contador);
    }

    private static int busquedaBinaria(String[] arreglo, String objetivo, int inicio, int fin, int nivel, Contador contador) {
        contador.profundidadMaxima = Math.max(contador.profundidadMaxima, nivel);
        if (inicio > fin) {
            return -1;
        }
        int medio = (inicio + fin) >>> 1;
        contador.comparaciones++;
        int comparacion = arreglo[medio].compareToIgnoreCase(objetivo);
        if (comparacion == 0) {
            return medio;
        }
        if (comparacion > 0) {
            return busquedaBinaria(arreglo, objetivo, inicio, medio - 1, nivel + 1, contador);
        }
        return busquedaBinaria(arreglo, objetivo, medio + 1, fin, nivel + 1, contador);
    }

    private static void imprimirResumen(ResultadoBusqueda resultado, String arregloMuestra) {
        System.out.println("================ RESULTADOS ================");
        System.out.println("Arreglo usado (primeros elementos): " + arregloMuestra);
        System.out.println("Encontrado?: " + (resultado.encontrado ? "SI" : "NO"));
        System.out.println("Posicion: " + resultado.posicion);
        System.out.println("Comparaciones: " + resultado.comparaciones);
        System.out.println("Profundidad max. (recursion): " + resultado.profundidadMaxima);
        System.out.println("Tiempo (ms): " + FORMAT.format(resultado.tiempoNs / 1_000_000.0));
        System.out.println("Memoria estimada (bytes): " + resultado.memoriaBytes);
        System.out.println("============================================");
    }

    private static int[] generarEnterosOrdenados(int cantidad) {
        int[] datos = new int[cantidad];
        int valor = RANDOM.nextInt(10);
        for (int i = 0; i < cantidad; i++) {
            valor += RANDOM.nextInt(5) + 1;
            datos[i] = valor;
        }
        return datos;
    }

    private static String[] generarCadenasOrdenadas(int cantidad) {
        String[] datos = new String[cantidad];
        for (int i = 0; i < cantidad; i++) {
            char inicial = (char) ('A' + RANDOM.nextInt(26));
            datos[i] = inicial + String.format("%03d", RANDOM.nextInt(900) + 100);
        }
        return datos;
    }

    private static int parseEntero(String valor) {
        try {
            return Integer.parseInt(valor);
        } catch (NumberFormatException error) {
            return 42; // neutral fallback
        }
    }

    private static final class Contador {
        int comparaciones;
        int profundidadMaxima;
    }
}
